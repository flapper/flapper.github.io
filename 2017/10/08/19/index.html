mysql (mariadb) root 비번 바꾸기 및 원격접속 허용

October 08 2017

Mariadb 10.1.11 에서 잘됨
SET PASSWORD FOR 'root'@'%' = PASSWORD('비밀번호');

INSERT INTO mysql.user (host,user,password) VALUES ('%','root',password('비밀번호')); GRANT ALL PRIVILEGES ON *.* TO 'root'@'%';
FLUSH PRIVILEGES;
